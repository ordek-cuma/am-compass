#
# Setup (one time):
#   cd spike
#   python3 -m venv .venv
#   ./.venv/bin/pip install -r requirements-render.txt
#   ./.venv/bin/python -m playwright install chromium
#
# The crawler auto-detects spike/.venv and shells out to it; if absent, JS sites are skipped
# with a warning and the static crawl still runs. .venv is gitignored.
